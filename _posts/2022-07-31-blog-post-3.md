    * 侧重Linux
* 自定Startup code(Entry-Point Symbol)
  * windows (use VC 6.0)

        #include <windows.h>
        int MyStartup(void)
        {
            int a = 10;
            HANDLE crtHeap = HeapCreate(HEAP_NO_SERIALIZE, 0x010, 4000 * 1024);
            int *p = (int *)HeapAlloc(crtHeap, HEAP_ZERO_MEMORY, 0x010);
            int i, j;
            for (i = 0; i < 100; i++)
            {
                // int j;
                for (j = 0; j < 100; j++, p++)
                {
                    *p = i * 100 + (j + 1);
                }
            }

            MessageBoxA(NULL, p, "abcd", MB_OK);
            return 0;
        }
